def get_tx(hash)
  if $use_coinbase_bbe && !$testnet
    url = "https://coinbase.com/network/tx/%s.json" % [hash]
  else
    url = "http://blockexplorer.com/%srawtx/%s" % [$testnet ? 'testnet/' : '',  hash]
  end
  json = open(url).read
  Bitcoin::Protocol::Tx.from_json(json)
rescue Exception
  nil
end

def run
  begin
    Timeout.timeout(T) do
      #@store.import("#{ENV['HOME']}/.bitcoin/testnet3/blocks")
      @store.import("bench.dat")
    end
  rescue Timeout::Error
    depth = @store.get_depth
    case @store.db.adapter_scheme
    when :sqlite
      size = File.size("test.db")
    when :postgres
      size = @store.db.fetch("select pg_database_size('bitcoin_test')").first[:pg_database_size]
    when :mysql
      size = @store.db.fetch("select sum(data_length+index_length) from information_schema.tables where table_schema = 'bitcoin_test';").first.to_a[0][1].to_i
    end
    puts "#{size.to_s.rjust(10)} | #{depth.to_s.rjust(5)} |"
    return [depth, size]
  rescue Exception
    puts "Error"
    p $!
    return [0, 0]
  end
end
